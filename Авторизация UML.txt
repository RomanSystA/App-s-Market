@startuml
actor Пользователь as user

participant WebAPI as front
participant BackEnd as back

database База_данных as db

user -> front: авторизовывается, вводя email/password/login/telegramID
activate user
activate front
front -> back: POST/user123
activate back
back -> db: проверяет, есть ли user_id=123
activate db
alt Пользователь есть
db --> back: user_id=123
deactivate db
back->back: проверка пароля
alt Верный пароль
back-->front: 200 OK, user_id=123
front-->front: подсвечивает окно пароля зеленым
else Неверный пароль
back-->front: 400 Неверный пароль
front-->front: подсвечивает окно пароля красным
front-->user: "Неверный пароль"
end alt
back --> front: 200 OK, user_id=123
front --> user: Пользователь авторизован
else Такого пользователя не существует
back --> front: 400 Такого пользователя не существует
front --> user: Такого пользователя не существует
else Данные введены неверно
back --> front: 400 Данные введены неверно
deactivate back
front --> user: Данные введены неверно
deactivate user
deactivate front
end alt
