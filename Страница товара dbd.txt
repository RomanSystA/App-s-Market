  id integer [pk]
  email varchar
  phone_number varchar
  tg_username varchar [note: 'логин в тг, к которому привязан акк']
}

 id integer [pk]
 seller_id integer
 name varchar
 category varchar
 base_price decimal
 rating smallient
 reviews_quantity decimal
}

}

  item_id integer

  rating_id smalliant [note: '1...5']

}

}
Table cart{
